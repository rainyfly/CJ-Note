class RadixCache(BasePrefixCache):
    def __init__(self, params: CacheInitParams):
        self.page_size = params.page_size
        self.eviction_policy = params.eviction_policy.lower()

        # 根据 page_size 选择键匹配函数
        if self.page_size == 1:
            self.key_match_fn = _key_match_page_size1
        else:
            self.key_match_fn = partial(_key_match_paged, page_size=self.page_size)

        # 根据策略名初始化驱逐策略
        if self.eviction_policy == "lru":
            self.eviction_strategy = LRUStrategy()
        elif self.eviction_policy == "lfu":
            self.eviction_strategy = LFUStrategy()
        # ... fifo, mru, filo, priority
        self.reset()

2.4 树遍历匹配实现

def _match_prefix_helper(self, node, key):
    access_time = time.monotonic()
    node.last_access_time = access_time
    child_key = self.get_child_key_fn(key)
    value = []

    while len(key) > 0 and child_key in node.children.keys():
        child = node.children[child_key]
        child.last_access_time = access_time
        prefix_len = self.key_match_fn(child.key, key)

        if prefix_len < len(child.key):
            # 部分匹配 -> 分裂节点
            new_node = self._split_node(child.key, child, prefix_len)
            value.append(new_node.value)
            node = new_node
            break
        else:
            # 完全匹配 -> 继续向下
            value.append(child.value)
            node = child
            key = key[prefix_len:]
            if len(key):
                child_key = self.get_child_key_fn(key)

    return value, node

3.4 哈希计算

compute_node_hash_values 使用 SHA256 链式哈希为每个页面生成位置感知的唯一标识：

def compute_node_hash_values(node, page_size):
    hash_values = []
    parent_hash = None
    if node.parent and node.parent.hash_value:
        parent_hash = node.parent.hash_value[-1]  # 父节点最后一页的哈希

    for start in range(0, len(node.key), page_size):
        page_tokens = node.key.token_ids[start : start + page_size]
        hash_val = get_hash_str(page_tokens, prior_hash=parent_hash)
        hash_values.append(hash_val)
        parent_hash = hash_val  # 链式传递
    return hash_values

6.2 TokenToKVPoolAllocator (page_size=1)

最简单的分配器，每个 token 独立占用一个 KV 缓存槽位。

class TokenToKVPoolAllocator(BaseTokenToKVPoolAllocator):
    def alloc(self, need_size):
        if self.need_sort and need_size > len(self.free_pages):
            self.merge_and_sort_free()  # 合并 release_pages 并排序
        if need_size > len(self.free_pages):
            return None
        select_index = self.free_pages[:need_size]
        self.free_pages = self.free_pages[need_size:]
        return select_index

    def free(self, free_index):
        if self.need_sort:
            self.release_pages = torch.cat((self.release_pages, free_index))
        else:
            self.free_pages = torch.cat((self.free_pages, free_index))

6.4 free_group 批量释放优化

延迟释放机制：通过 free_group_begin() 和 free_group_end() 包裹的代码块中，所有 free() 调用不会立即归还内存，而是暂存到 free_group 列表。在 free_group_end() 时一次性 torch.cat 后释放，减少内存碎片和 Tensor 拼接开销。

6.5 need_sort 双缓冲区设计

缓冲区用途说明
free_pages可直接分配的页有序排列，从前端分配
release_pages待合并的释放页释放时追加，分配不足时合并排序

当 need_sort=True 时，释放的页先进入 release_pages，仅在 free_pages 不足时才执行 merge_and_sort_free() 合并排序，减少排序频率。

7.4 lock_ref 保护机制

def inc_lock_ref(self, node):
    """锁定节点及其所有祖先，防止被驱逐"""
    delta = 0
    while node != self.root_node:
        if node.lock_ref == 0:
            self.evictable_size_ -= len(node.key)
            self.protected_size_ += len(node.key)
        node.lock_ref += 1
        node = node.parent
    return delta

def dec_lock_ref(self, node):
    """解锁节点及其所有祖先"""
    while node != self.root_node:
        if node.lock_ref == 1:
            self.evictable_size_ += len(node.key)
            self.protected_size_ -= len(node.key)
        node.lock_ref -= 1
        node = node.parent
驱逐安全性：当请求正在使用某个前缀缓存时，inc_lock_ref 会锁定从当前节点到根节点的整条路径。只有路径上所有引用都释放(lock_ref 回到 0)后，节点才可被驱逐。_collect_leaves() 仅收集 lock_ref == 0 的叶节点。

8.5 加载回填 (load_back)

def load_back(self, node, mem_quota=None):
    # 收集从当前驱逐节点到最近活跃祖先的所有待加载节点
    nodes_to_load = []
    while node.evicted:
        nodes_to_load.insert(0, node)
        node = node.parent
    ancestor_node = node

    # 保护祖先节点不被驱逐
    self.inc_lock_ref(ancestor_node)

    # 一次性加载所有 host_indices
    host_indices = torch.cat([n.host_value for n in nodes_to_load])
    device_indices = self.cache_controller.load(host_indices, node_id)

    # 分配 device 端索引到各节点
    offset = 0
    for node in nodes_to_load:
        node.value = device_indices[offset : offset + len(node.host_value)]
        offset += len(node.host_value)

    self.evictable_size_ += len(device_indices)
    return device_indices

8.7 预取 (Prefetch from Storage)

预取机制：当请求的 token 序列在外部存储中可能存在时，prefetch_from_storage 会提前将数据从外部存储加载到 Host 内存。支持三种停止策略：best_effort(尽力而为，立即可终止)、wait_complete(等待全部完成)、timeout(超时自动终止，超时时间随页数线性增长)。

class StorageBackendFactory:
    _registry: Dict[str, Dict[str, Any]] = {}

    @classmethod
    def register_backend(cls, name, module_path, class_name):
        """惰性注册：仅在实际创建时才 import 模块"""
        def loader():
            return cls._load_backend_class(module_path, class_name, name)

    @classmethod
    def create_backend(cls, backend_name, storage_config, mem_pool_host, **kwargs):
        if backend_name in cls._registry:

9.4 TP 同步

张量并行一致性：HiRadixCache 中所有修改树结构的操作(write_through 完成、load_back、prefetch 终止)都需要在 TP worker 间同步。通过 torch.distributed.all_reduce 使用 ReduceOp.MIN 确保所有 rank 看到一致的完成队列大小，保证树结构的一致性。

10.3 evict_from_tree_cache

def evict_from_tree_cache(tree_cache, num_tokens):
    if tree_cache.is_chunk_cache():
        return  # ChunkCache 不使用树驱逐

    allocator = tree_cache.token_to_kv_pool_allocator

    if hasattr(allocator, "full_available_size"):
        # 混合分配器(SWA): 分别驱逐 full 和 swa 池
        full_num = max(0, num_tokens - allocator.full_available_size())
        swa_num = max(0, num_tokens - allocator.swa_available_size())
        tree_cache.evict(full_num, swa_num)
    else:
        # 标准分配器
        if allocator.available_size() < num_tokens:
            tree_cache.evict(num_tokens)

11.2 cache_finished_req 详解

def cache_finished_req(self, req, is_insert=True):
    kv_committed_len = req.pop_committed_kv_cache()
    token_ids = (req.origin_input_ids + req.output_ids)[:kv_committed_len]
    kv_indices = self.req_to_token_pool.req_to_token[req.req_pool_idx, :len(token_ids)]

    keys = self._page_align_keys(req.fill_ids)
    values = kv_indices[:len(keys)].to(dtype=torch.int64, copy=True)
    radix_key = RadixKey(keys, req.extra_key)

    if is_insert:
        new_prefix_len = self.insert(radix_key, values, priority=req.priority)
        # 释放已在树中存在的重复索引
        self.token_to_kv_pool_allocator.free(
            kv_indices[req.cache_protected_len : new_prefix_len]
        )

    # 释放未对齐的尾部
    self.token_to_kv_pool_allocator.free(kv_indices[len(keys):])
    # 释放请求槽位并解锁
    self.req_to_token_pool.free(req.req_pool_idx)
    self.dec_lock_ref(req.last_node)

12. 总结

SGLang 缓存系统核心设计要点

  • Radix Tree 前缀缓存：通过共享公共前缀节点，实现 O(n) 的前缀匹配和高效的 KV 缓存复用，支持节点分裂以精确匹配任意长度前缀
  • 三级内存池层次：ReqToTokenPool(请求映射) -> TokenToKVPoolAllocator(索引管理) -> KVCache(物理存储)，各司其职，层次分明
  • 多种 KVCache 类型：MHATokenToKVPool(标准K/V分离)、MLATokenToKVPool(MLA联合压缩)、FP4变体、SWA双池 等针对不同模型架构的优化实现
  • 分页分配器：PagedTokenToKVPoolAllocator 通过 Triton kernel(alloc_extend_kernel/alloc_decode_kernel)实现高效的三段式分页分配
  • 6 种驱逐策略：LRU/LFU/FIFO/MRU/FILO/Priority 可插拔设计，基于最小堆的驱逐执行支持级联(父节点自动加入堆)
  • lock_ref 保护机制：路径级引用计数确保正在使用的前缀缓存不会被驱逐，安全与效率并重
  • HiRadixCache 分层缓存：GPU + Host + External Storage 三级存储层次，支持 write_through/write_back 写策略和异步加载回填
  • 7 种外部存储后端：file/nixl/mooncake/hf3fs/aibrix/eic/dynamic，通过工厂模式惰性加载，支持自定义后端动态注册
  • TP 一致性保障：HiRadixCache 通过 all_reduce 同步确保所有 TP rank 的树结构一致
  • EAGLE bigram 支持：通过 RadixKey 的 is_bigram 标志和 convert_to_bigram_key 转换，适配 EAGLE 投机解码的特殊缓存键格式
  • SHA256 链式哈希：compute_node_hash_values 为每个页面生成位置感知的哈希标识，支持跨实例的缓存识别和外部存储索引
